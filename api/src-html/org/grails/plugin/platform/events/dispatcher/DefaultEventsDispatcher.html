001    package org.grails.plugin.platform.events.dispatcher;
002    
003    import groovy.lang.Closure;
004    import org.codehaus.groovy.grails.commons.GrailsClassUtils;
005    import org.grails.plugin.platform.events.EventObject;
006    import org.grails.plugin.platform.events.dispatcher.builder.EventsMethodBuilder;
007    import org.grails.plugin.platform.events.dispatcher.builder.MappedEventMethod;
008    
009    import java.util.HashMap;
010    import java.util.List;
011    import java.util.Map;
012    
013    /**
014     * @author Stephane Maldini <smaldini@doc4web.com>
015     * @version 1.0
016     * @file
017     * @date 16/01/12
018     * @section DESCRIPTION
019     * <p/>
020     * [Does stuff]
021     */
022    public class DefaultEventsDispatcher implements EventsDispatcher {
023    
024        public final static String PROPERTY_NAME = "events";
025    
026        public Map<String, Map<String, MappedEventMethod>> getMappedMethodsByClass() {
027            return mappedMethodsByClass;
028        }
029    
030        private Map<String, Map<String, MappedEventMethod>> mappedMethodsByClass =
031                new HashMap<String, Map<String, MappedEventMethod>>();
032    
033        public boolean when(EventObject event, String className, String methodName) {
034            return false;
035        }
036    
037        public void scanClassForMappings(Class<?> clazz) {
038            EventsMethodBuilder delegate = new EventsMethodBuilder(clazz);
039    
040            Closure c = (Closure) GrailsClassUtils.getStaticPropertyValue(clazz, PROPERTY_NAME);
041    
042            if (c != null) {
043                c = (Closure) c.clone();
044                c.setResolveStrategy(Closure.DELEGATE_ONLY);
045                c.setDelegate(delegate);
046                c.call();
047    
048                mappedMethodsByClass.put(clazz.getName(), delegate.getMappedProperties());
049            }
050        }
051    
052        public void scanClassesForMappings(List<Class<?>> classes) {
053            for (Class<?> clazz : classes) {
054                scanClassForMappings(clazz);
055            }
056        }
057    }
