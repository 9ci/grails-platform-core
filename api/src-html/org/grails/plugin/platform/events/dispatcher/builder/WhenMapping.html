001    package org.grails.plugin.platform.events.dispatcher.builder;
002    
003    import groovy.lang.Closure;
004    
005    /**
006    * @author Stephane Maldini <smaldini@doc4web.com>
007    * @version 1.0
008    * @file
009    * @date 17/01/12
010    * @section DESCRIPTION
011    * <p/>
012    * [Does stuff]
013    */
014    public class WhenMapping extends EventMapping {
015    
016        private Closure when;
017    
018        @Override
019        public void setParameter(Object mappedParameter) {
020            if (!(mappedParameter instanceof Closure)) {
021                throw new IllegalArgumentException("Parameter for mapping [" +
022                        MappedEventMethod.WHEN_MAPPING + "] of property [" +
023                        getMethodName() + "] of class [" + getOwningClass() + "] must be a Closure");
024            }
025    
026            when = (Closure) mappedParameter;
027    
028            Class<?>[] params = when.getParameterTypes();
029            if (params.length != 1) {
030                throw new IllegalArgumentException("Parameter for mapping [" + MappedEventMethod.WHEN_MAPPING + "] of property [" + getMethodName() + "] of class [" + getOwningClass() + "] must be a Closure taking 1 parameter (event)");
031            }
032    
033            super.setParameter(mappedParameter);
034        }
035    }
