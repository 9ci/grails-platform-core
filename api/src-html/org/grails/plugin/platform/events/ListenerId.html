018    package org.grails.plugin.platform.events;
019    
020    import groovy.lang.Closure;
021    import org.grails.plugin.platform.events.registry.EventsRegistry;
022    
023    import java.io.Serializable;
024    import java.lang.reflect.Method;
025    import java.util.regex.Matcher;
026    import java.util.regex.Pattern;
027    
028    /**
029     * @author Stephane Maldini <smaldini@doc4web.com>
030     * @version 1.0
031     * @file
032     * @date 09/01/12
033     * @section DESCRIPTION
034     * <p/>
035     * [Does stuff]
036     */
037    public class ListenerId implements Serializable {
038        private static final String CLOSURE_METHOD_NAME = "call";
039        private static final String ID_CLASS_SEPARATOR = ":";
040        private static final String ID_METHOD_SEPARATOR = "#";
041        private static final String ID_HASHCODE_SEPARATOR = "@";
042        private static final Pattern idRegex = Pattern.compile(
043                "([^" + ID_CLASS_SEPARATOR + "]*)?" +
044                        "(" + ID_CLASS_SEPARATOR + "([^" + ID_METHOD_SEPARATOR + "]*))?"
045                        + "(" + ID_METHOD_SEPARATOR + "([^" + ID_HASHCODE_SEPARATOR + "]*))?"
046                        + "(" + ID_HASHCODE_SEPARATOR + "(-?\\d*))?");
047    
048        private String className;
049        private String methodName;
050        private String hashCode;
051        private String topic;
052    
053        public ListenerId(String topic) {
054            this(topic, null, null, null);
055        }
056    
057        public ListenerId(String topic, String className, String methodName, String hashCode) {
058            this.className = className;
059            this.methodName = methodName;
060            this.hashCode = hashCode;
061            if (topic != null && !topic.isEmpty()) {
062                this.topic = topic.startsWith(EventsRegistry.GRAILS_TOPIC_PREFIX) ?
063                        topic :
064                        EventsRegistry.GRAILS_TOPIC_PREFIX + topic;
065            }
066        }
067    
068        public String getTopic() {
069            return topic;
070        }
071    
072        public void setTopic(String topic) {
073            this.topic = topic;
074        }
075    
076        public String getClassName() {
077            return className;
078        }
079    
080        public void setClassName(String className) {
081            this.className = className;
082        }
083    
084        public String getMethodName() {
085            return methodName;
086        }
087    
088        public void setMethodName(String methodName) {
089            this.methodName = methodName;
090        }
091    
092        public String getHashCode() {
093            return hashCode;
094        }
095    
096        public void setHashCode(String hashCode) {
097            this.hashCode = hashCode;
098        }
099    
100        public String toString() {
101            return (topic != null ? topic : "") + (className != null ? ID_CLASS_SEPARATOR + className : "")
102                    + (methodName != null ? ID_METHOD_SEPARATOR + methodName : "")
103                    + (hashCode != null ? ID_HASHCODE_SEPARATOR + hashCode : "");
104        }
105    
106        static public ListenerId build(String topic, Object target, Method callback) {
107            return new ListenerId(topic, target.getClass().getName(), callback.getName(), Integer.toString(target.hashCode()));
108        }
109    
110        static public ListenerId build(String topic, Closure target) {
111            return new ListenerId(topic, target.getClass().getName(), CLOSURE_METHOD_NAME, Integer.toString(target.hashCode()));
112        }
113    
114        static public ListenerId parse(String id) {
115            Matcher parsed = idRegex.matcher(id);
116            if (parsed.matches())
117                return new ListenerId(parsed.group(1), parsed.group(3), parsed.group(5), parsed.group(7));
118            else
119                return null;
120        }
121    
122        public boolean matches(ListenerId listener) {
123            Boolean result = null;
124            if (this.topic != null) {
125                result = this.topic.equals(listener.getTopic());
126            }
127            if (this.className != null) {
128                result = result == null || result;
129                result &= this.className.equals(listener.getClassName());
130                if (this.methodName != null) {
131                    result &= this.methodName.equals(listener.getMethodName());
132                    if (this.hashCode != null) {
133                        result &= this.hashCode.equals(listener.getHashCode());
134                    }
135                }
136            }
137    
138            return result != null && result;
139        }
140    
141        public boolean equals(String patternId) {
142            ListenerId listener = parse(patternId);
143            return matches(listener);
144        }
145    
146        @Override
147        public boolean equals(Object o) {
148            if (this == o) return true;
149            if (o == null || getClass() != o.getClass()) return false;
150    
151            ListenerId listener = (ListenerId) o;
152    
153            return !(className != null ? !className.equals(listener.className) : listener.className != null) &&
154                    !(hashCode != null ? !hashCode.equals(listener.hashCode) : listener.hashCode != null) &&
155                    !(methodName != null ? !methodName.equals(listener.methodName) : listener.methodName != null) &&
156                    !(topic != null ? !topic.equals(listener.topic) : listener.topic != null);
157    
158        }
159    
160        @Override
161        public int hashCode() {
162            int result = className != null ? className.hashCode() : 0;
163            result = 31 * result + (methodName != null ? methodName.hashCode() : 0);
164            result = 31 * result + (hashCode != null ? hashCode.hashCode() : 0);
165            result = 31 * result + (topic != null ? topic.hashCode() : 0);
166            return result;
167        }
168    }
