024        public final static String WHEN_MAPPING = "when";
025        protected static Map<String, List<Class<? extends EventMapping>>> mappings = new HashMap<String, List<Class<? extends EventMapping>>>();
026    
027        static {
028            mappings.put(WHEN_MAPPING, new ArrayList<Class<? extends EventMapping>>() {{
029                add(WhenMapping.class);
030            }});
031        }

033        protected static final Log LOG = LogFactory.getLog(MappedEventMethod.class);
034

037        protected String methodName;
038        protected Method methodNode;
039    
040        protected Map<String, EventMapping> appliedMappings = new LinkedHashMap<String, EventMapping>();
041        protected Class<?> owningClass;

043    
044        private int order; // what order to property appears in
045        @SuppressWarnings("rawtypes")
046        private Map attributes = Collections.EMPTY_MAP; // a map of attributes of property
047        private Map<String, Object> metaMappings = new HashMap<String, Object>();

049        public MappedEventMethod(Class<?> clazz, String methodName) {
050            this(clazz, methodName, null);
051        }
052        public MappedEventMethod(Class<?> clazz, String methodName, Method methodNode) {
053            owningClass = clazz;
054            this.methodName = methodName;
055            this.methodNode = methodNode;
056            bean = new BeanWrapperImpl(this);
057        }
058    
059        public void setMethodNode(Method methodNode) {
060            this.methodNode = methodNode;
061        }
062    
063        public static void removeMapping(String name, Class mappingClass) {
064            Assert.hasLength(name, "Argument [name] cannot be null");
065    
066            List<Class<? extends EventMapping>> objects = getOrInitializeMapping(name);
067            objects.remove(mappingClass);
068        }
069    
070        public static void removeMapping(String name) {
071            Assert.hasLength(name, "Argument [name] cannot be null");
072    
073            List<Class<? extends EventMapping>> objects = getOrInitializeMapping(name);
074            objects.clear();
075        }
076    
077        public static void registerNewMapping(String name, Class<? extends EventMapping> mappingClass) {
078            Assert.hasLength(name, "Argument [name] cannot be null");
079            if (mappingClass == null || !EventMapping.class.isAssignableFrom(mappingClass)) {
080                throw new IllegalArgumentException("Argument [mappingClass] with value [" + mappingClass +
081                        "] is not a valid mapping");
082            }
083    
084            List<Class<? extends EventMapping>> objects = getOrInitializeMapping(name);
085            objects.add(mappingClass);
086        }
087    
088        private static List<Class<? extends EventMapping>> getOrInitializeMapping(String name) {
089            List<Class<? extends EventMapping>> objects = mappings.get(name);
090            if (objects == null) {
091                objects = new ArrayList<Class<? extends EventMapping>>();
092                mappings.put(name, objects);
093            }
094            return objects;
095        }
096    
097    
098        public static boolean hasRegisteredMapping(String mappingName) {
099            return mappings.containsKey(mappingName);
100        }

158        /**
159         * Checks with this MappedMethod instance supports applying the specified mapping.
160         *
161         * @param mappingName The name of the mapping
162         * @return True if the mapping is supported
163         */
164        public boolean supportsMapping(String mappingName) {
165    
166            if (!mappings.containsKey(mappingName)) {
167                return bean.isWritableProperty(mappingName);
168            }
169    
170            try {
171                EventMapping c = instantiateMapping(mappingName, false);
172                return c != null && c.supports(methodNode);
173            } catch (Exception e) {
174                e.printStackTrace();
175                LOG.error("Exception thrown instantiating mapping [" + mappingName +
176                        "] to class [" + owningClass + "]", e);
177                throw new RuntimeException("Exception thrown instantiating  mapping [" + mappingName +
178                        "] to class [" + owningClass + "]");
179            }
180        }
181    
182        /**
183         * Applies a mapping for the specified name and consraint value.
184         *
185         * @param mappingName       The name of the mapping
186         * @param constrainingValue The constraining value
187         * @throws RuntimeException Thrown when the specified mapping is not supported by this ConstrainedProperty. Use <code>supportsMapping(String mappingName)</code> to check before calling
188         */
189        public void applyMapping(String mappingName, Object constrainingValue) {
190    
191            if (mappings.containsKey(mappingName)) {
192                if (constrainingValue == null) {
193                    appliedMappings.remove(mappingName);
194                } else {
195                    try {
196                        EventMapping c = instantiateMapping(mappingName, true);
197                        if (c != null) {
198                            c.setParameter(constrainingValue);
199                            appliedMappings.put(mappingName, c);
200                        }
201                    } catch (Exception e) {
202                        LOG.error("Exception thrown applying mapping [" + mappingName +
203                                "] to class [" + owningClass + "] for value [" + constrainingValue + "]: " + e.getMessage(), e);
204                        throw new RuntimeException("Exception thrown applying mapping [" + mappingName +
205                                "] to class [" + owningClass + "] for value [" + constrainingValue + "]: " + e.getMessage(), e);
206                    }
207                }
208            } else if (bean.isWritableProperty(mappingName)) {
209                bean.setPropertyValue(mappingName, constrainingValue);
210            } else {
211                throw new RuntimeException("Mapping [" + mappingName + "] is not supported for property [" +
212                        methodName + "] of class [" + owningClass + "] with type [" + methodNode + "]");
213            }
214        }
215    
216        private EventMapping instantiateMapping(String mappingName, boolean validate) throws InstantiationException, IllegalAccessException {
217            List<Class<? extends EventMapping>> candidateMappings = mappings.get(mappingName);
218    
219            for (Class<? extends EventMapping> mappingFactory : candidateMappings) {
220                EventMapping m = mappingFactory.newInstance();
221    
222                m.setOwningClass(owningClass);
223                m.setMethodName(methodName);
224    
225                if (validate && m.isValid()) {
226                    return m;
227                } else if (!validate) {
228                    return m;
229                }
230    
231            }
232            return null;
233        }
234    
235        /* (non-Javadoc)
236        * @see java.lang.Object#toString()
237        */
238        @Override
239        public String toString() {
240            return new ToStringBuilder(this)
241                    .append(owningClass)
242                    .append(methodName)
243                    .append(methodNode)
244                    .append(appliedMappings)
245                    .toString();
246        }

255            metaMappings.put(name, value);
256        }

266        }
