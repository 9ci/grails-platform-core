001    package org.grails.plugin.platform.events.dispatcher.builder;
002    
003    import grails.util.GrailsUtil;
004    import groovy.lang.GroovySystem;
005    import groovy.lang.MetaClass;
006    import groovy.lang.MissingMethodException;
007    import groovy.lang.MissingPropertyException;
008    import groovy.util.BuilderSupport;
009    import org.grails.plugin.platform.events.Listener;
010    import org.springframework.beans.InvalidPropertyException;
011    import org.springframework.util.ReflectionUtils;
012    
013    import java.lang.reflect.Method;
014    import java.util.Collections;
015    import java.util.LinkedHashMap;
016    import java.util.Map;
017    
018    /**
019     * @author Stephane Maldini <smaldini@doc4web.com>
020     * @version 1.0
021     * @file
022     * @date 17/01/12
023     * @section DESCRIPTION
024     * <p/>
025     * [Does stuff]
026     */
027    public class EventsMethodBuilder extends BuilderSupport {
028        private Map<String, MappedEventMethod> mappedProperties = new LinkedHashMap<String, MappedEventMethod>();
029        private int order = 1;
030        private Class<?> targetClass;
031        private MetaClass targetMetaClass;
032    
033        public EventsMethodBuilder(Object target) {
034            this(target.getClass());
035        }
036    
037        public EventsMethodBuilder(Class<?> targetClass) {
038            this.targetClass = targetClass;
039            this.targetMetaClass = GroovySystem.getMetaClassRegistry().getMetaClass(targetClass);
040        }
041    
042    
043        @Override
044        protected Object doInvokeMethod(String methodName, Object name, Object args) {
045            try {
046                return super.doInvokeMethod(methodName, name, args);
047            } catch (MissingMethodException e) {
048                return targetMetaClass.invokeMethod(targetClass, methodName, args);
049            }
050        }
051    
052        @Override
053        public Object getProperty(String property) {
054            try {
055                return super.getProperty(property);
056            } catch (MissingPropertyException e) {
057                return targetMetaClass.getProperty(targetClass, property);
058            }
059        }
060    
061        @Override
062        public void setProperty(String property, Object newValue) {
063            try {
064                super.setProperty(property, newValue);
065            } catch (MissingPropertyException e) {
066                targetMetaClass.setProperty(targetClass, property, newValue);
067            }
068    
069        }
070    
071        @SuppressWarnings("rawtypes")
072        @Override
073        protected Object createNode(Object name, final Map attributes) {
074            // we do this so that missing property exception is throw if it doesn't exist
075    
076            try {
077                final String method = (String) name;
078                MappedEventMethod mm;
079                if (mappedProperties.containsKey(method)) {
080                    mm = mappedProperties.get(method);
081                } else {
082                    mm = new MappedEventMethod(targetClass, method);
083                    final MappedEventMethod _mm = mm;
084                    ReflectionUtils.doWithMethods(targetClass, new ReflectionUtils.MethodCallback() {
085                        public void doWith(Method _methodNode) throws IllegalArgumentException, IllegalAccessException {
086                            if (_methodNode.getName().equals(method)) {
087                                if (_methodNode.getAnnotation(Listener.class) == null) {
088                                    throw new RuntimeException(targetClass.getName() + "#" + _methodNode.getName() + " is not" +
089                                            "a @Listener");
090                                }
091                                if (_methodNode == null) {
092                                    throw new MissingMethodException(method, targetClass, new Object[]{attributes}, true);
093                                }
094    
095                                _mm.setMethodNode(_methodNode);
096                                _mm.setOrder(order++);
097                                mappedProperties.put(method, _mm);
098                                return;
099                            }
100                        }
101                    });
102    
103                }
104    
105                if (mm.getMethodNode() == null) {
106                    return mm;
107                }
108    
109                for (Object o : attributes.keySet()) {
110                    String mappingName = (String) o;
111                    final Object value = attributes.get(mappingName);
112                    if (mm.supportsMapping(mappingName)) {
113                        mm.applyMapping(mappingName, value);
114                    } else {
115                        if (MappedEventMethod.hasRegisteredMapping(mappingName)) {
116                            // constraint is registered but doesn't support this property's type
117                            GrailsUtil.warn("Method [" + mm.getMethodName() + "] of domain class " +
118                                    targetClass.getName() + " has type [" + mm.getMethodNode().getName() +
119                                    "] and doesn't support mapping [" + mappingName +
120                                    "]. This mapping will not be checked.");
121                        } else {
122                            // in the case where the constraint is not supported we still retain meta data
123                            // about the constraint in case its needed for other things
124                            mm.addMetaMapping(mappingName, value);
125                        }
126                    }
127                }
128    
129                return mm;
130            } catch (InvalidPropertyException ipe) {
131                throw new MissingMethodException((String) name, targetClass, new Object[]{attributes});
132            }
133        }
134    
135        @SuppressWarnings("rawtypes")
136        @Override
137        protected Object createNode(Object name, Map attributes, Object value) {
138            throw new MissingMethodException((String) name, targetClass, new Object[]{attributes, value});
139        }
140    
141        @Override
142        protected void setParent(Object parent, Object child) {
143            // do nothing
144        }
145    
146        @Override
147        protected Object createNode(Object name) {
148            return createNode(name, Collections.EMPTY_MAP);
149        }
150    
151        @Override
152        protected Object createNode(Object name, Object value) {
153            return createNode(name, Collections.EMPTY_MAP, value);
154        }
155    
156        public Map<String, MappedEventMethod> getMappedProperties() {
157            return mappedProperties;
158        }
159    }
